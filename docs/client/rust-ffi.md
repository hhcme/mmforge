## 桥接架构

```
┌─────────────────────┐
│   Swift 层          │

           │ 调用
┌──────────▼──────────┐
│   C ABI 层          │

           │ FFI
┌──────────▼──────────┐
```

## Rust 侧：导出 C API

```rust
// crates/mmforge-core/src/ffi.rs

use std::ffi::{CStr, CString};
use std::os::raw::c_char;
use std::ptr;

/// 不透明句柄
pub struct ModelHandle {
    inner: Box<LsmModel>,
}

/// C 兼容的模型信息
#[repr(C)]
pub struct CModelInfo {
    pub source_format: u32,
    pub name: *mut c_char,
    pub faces: u32,
    pub edges: u32,
    pub vertices: u32,
    pub bbox_min: [f64; 3],
    pub bbox_max: [f64; 3],
}

/// 打开文件
#[no_mangle]
pub extern "C" fn mmforge_open_file(path: *const c_char) -> *mut ModelHandle {
    let path = unsafe { CStr::from_ptr(path) }.to_str().unwrap();

    match open_file_internal(path) {
        Ok(model) => Box::into_raw(Box::new(ModelHandle { inner: Box::new(model) })),
        Err(_) => ptr::null_mut(),
    }
}

/// 关闭文件
#[no_mangle]
pub extern "C" fn mmforge_close_file(handle: *mut ModelHandle) {
    if !handle.is_null() {
        unsafe { drop(Box::from_raw(handle)); }
    }
}

/// 获取模型信息
#[no_mangle]
pub extern "C" fn mmforge_get_model_info(handle: *mut ModelHandle) -> CModelInfo {
    let handle = unsafe { &*handle };
    let info = handle.inner.info();

    CModelInfo {
        source_format: info.source_format as u32,
        name: CString::new(info.name.clone()).unwrap().into_raw(),
        faces: info.faces,
        edges: info.edges,
        vertices: info.vertices,
        bbox_min: info.bbox_min,
        bbox_max: info.bbox_max,
    }
}

/// 获取渲染数据
#[no_mangle]
pub extern "C" fn mmforge_get_render_data(
    handle: *mut ModelHandle,
    options: CTessellationOptions,
) -> CRenderData {
    let handle = unsafe { &*handle };
    let data = handle.inner.get_render_data(&options.into());

    CRenderData {
        positions: data.positions.as_ptr(),
        positions_len: data.positions.len(),
        normals: data.normals.as_ptr(),
        normals_len: data.normals.len(),
        indices: data.indices.as_ptr(),
        indices_len: data.indices.len(),
    }
}
```

## cbindgen 配置

```toml
# cbindgen.toml
language = "C"
include_guard = "MMFORGE_BRIDGE_H"
autogen_warning = "// Auto-generated by cbindgen. Do not edit."
style = "tag"
```

生成头文件：

```c
// mmforge_bridge.h
#ifndef MMFORGE_BRIDGE_H
#define MMFORGE_BRIDGE_H

#include <stdint.h>

typedef struct ModelHandle ModelHandle;

typedef struct {
    uint32_t source_format;
    char* name;
    uint32_t faces;
    uint32_t edges;
    uint32_t vertices;
    double bbox_min[3];
    double bbox_max[3];
} CModelInfo;

typedef struct {
    const float* positions;
    uintptr_t positions_len;
    const float* normals;
    uintptr_t normals_len;
    const uint32_t* indices;
    uintptr_t indices_len;
} CRenderData;

ModelHandle* mmforge_open_file(const char* path);
void mmforge_close_file(ModelHandle* handle);
CModelInfo mmforge_get_model_info(ModelHandle* handle);
CRenderData mmforge_get_render_data(ModelHandle* handle, CTessellationOptions options);

#endif
```

## Swift 桥接层

```swift
// RustFFI.swift

import Foundation

class RustBridge {
    /// 打开文件
    func openFile(path: String) throws -> ModelHandle {
        let handle = path.withCString { cPath in
            mmforge_open_file(cPath)
        }
        guard let handle = handle else {
            throw MMForgeError.openFailed(path)
        }
        return ModelHandle(ptr: handle)
    }

    /// 获取模型信息
    func getModelInfo(handle: ModelHandle) -> ModelInfo {
        let cInfo = mmforge_get_model_info(handle.ptr)
        return ModelInfo(
            name: String(cString: cInfo.name),
            faces: Int(cInfo.faces),
            edges: Int(cInfo.edges),
            vertices: Int(cInfo.vertices),
            boundingBox: BoundingBox(
                min: SIMD3(cInfo.bbox_min),
                max: SIMD3(cInfo.bbox_max)
            )
        )
    }

    /// 获取渲染数据（零拷贝）
    func getRenderData(handle: ModelHandle, options: TessellationOptions) -> RenderData {
        let cData = mmforge_get_render_data(handle.ptr, options.toC())

        // 零拷贝：直接引用 Rust 内存
        let positions = UnsafeBufferPointer(
            start: cData.positions,
            count: Int(cData.positions_len)
        )
        let normals = UnsafeBufferPointer(
            start: cData.normals,
            count: Int(cData.normals_len)
        )
        let indices = UnsafeBufferPointer(
            start: cData.indices,
            count: Int(cData.indices_len)
        )

        return RenderData(
            positions: Array(positions),
            normals: Array(normals),
            indices: Array(indices)
        )
    }
}
```

## 错误处理

```swift
enum MMForgeError: Error {
    case openFailed(String)
    case parseFailed(String)
    case tessellationFailed(String)
    case cancelled

    var localizedDescription: String {
        switch self {
        case .openFailed(let path):
            return "Failed to open file: \(path)"
        case .parseFailed(let reason):
            return "Parse failed: \(reason)"
        case .tessellationFailed(let reason):
            return "Tessellation failed: \(reason)"
        case .cancelled:
            return "Operation cancelled"
        }
    }
}
```

## 内存管理

### Rust 内存

```rust
// Rust 分配的内存由 Rust 管理
// Swift 通过句柄引用，不直接管理 Rust 内存

// 创建句柄
let handle = Box::into_raw(Box::new(ModelHandle { ... }));

// 释放句柄
unsafe { drop(Box::from_raw(handle)); }
```

### 零拷贝渲染数据

```swift
// 方案 1：复制到 Metal Buffer（简单但有拷贝开销）
let positions: [Float] = Array(bufferPointer)
let metalBuffer = device.makeBuffer(bytes: positions, ...)

// 方案 2：直接使用 Rust 内存（零拷贝，需要保证 Rust 内存有效）
let metalBuffer = device.makeBuffer(
    bytesNoCopy: UnsafeMutableRawPointer(mutating: cData.positions),
    length: Int(cData.positions_len) * MemoryLayout<Float>.size,
    ...
)
```
